export function assertEconomicValidityStrict(obj: Record<string, any>) {
  for (const [key, value] of Object.entries(obj)) {
    if (value === null || value === undefined) {
      throw new Error(`INVALID_NULL_${key}`);
    }
    if (typeof value === 'number') {
      if (!Number.isFinite(value)) throw new Error(`NON_FINITE_${key}`);
      if (value < 0 && key !== 'delta') throw new Error(`NEGATIVE_${key}`);
    }
  }
}
